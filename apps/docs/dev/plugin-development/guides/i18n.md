## Getting the User's Locale

```typescript
const locale = await sdk.workspace.getLocale();
// Returns: 'en', 'de', 'fr', etc.
```

## Providing Translations

```typescript
const translations = {
  en: { greeting: 'Hello!' },
  de: { greeting: 'Hallo!' },
  fr: { greeting: 'Bonjour!' },
};

const t = (key: string) => {
  return translations[locale]?.[key] ?? translations.en[key];
};

sdk.ui.registerSidebarPanel({
  title: t('greeting'),
  // ...
});
```
